# Importa as bibliotecas necessárias
import socket          # Comunicação em rede
import threading       # Permite atender vários clientes simultaneamente

# Número de bytes utilizados para armazenar o tamanho da mensagem
HEADER = 64

# Porta em que o servidor ficará escutando
PORT = 5050

# 0.0.0.0 permite aceitar conexões de qualquer endereço IP
SERVER = "0.0.0.0"

# Endereço do servidor
ADDR = (SERVER, PORT)

# Formato de codificação das mensagens
FORMAT = "utf-8"

# Mensagem utilizada para indicar desconexão
DISCONNECT_MESSAGE = "!DISCONNECT"

# Cria um socket TCP utilizando IPv4
server = socket.socket(socket.AF_INET, socket.SOCK_STREAM)

# Associa o socket ao endereço e porta
server.bind(ADDR)

# Coloca o servidor em modo de escuta
server.listen()

# Lista contendo todos os clientes conectados
clients = []

# Lista contendo os nicknames dos clientes
nicknames = []

# Remove um cliente do servidor
def remove_client(client):

    # Obtém a posição do cliente na lista
    index = clients.index(client)

    # Fecha a conexão
    client.close()

    # Exibe a lista atual de clientes
    print(clients)

    # Obtém o nickname correspondente
    nickname = nicknames[index]

    # Remove o cliente da lista
    clients.remove(client)

    # Informa no terminal que o cliente saiu
    print(f"{nickname} desconectou do chat")

    # Informa aos demais usuários que ele saiu
    broadcast(f"{nickname} left the chat".encode(FORMAT))

    # Remove o nickname da lista
    nicknames.remove(nickname)

# Envia uma mensagem para todos os clientes conectados
def broadcast(message):

    # Percorre todos os clientes
    for client in clients:

        # Envia a mensagem para cada cliente
        client.send(message)

# Função executada para cada cliente conectado
def handle_client(client, addr):

    connected = True

    while connected:

        try:

            # Recebe o tamanho da mensagem
            msg_length = client.recv(HEADER).decode(FORMAT)

            if msg_length:

                # Converte para inteiro
                msg_length = int(msg_length)

                # Recebe exatamente a quantidade de bytes informada
                msg = client.recv(msg_length).decode(FORMAT)

                # Verifica se é uma solicitação de desconexão
                if msg == DISCONNECT_MESSAGE:

                    remove_client(client)
                    break

                # Envia a mensagem para todos os clientes
                broadcast(msg.encode(FORMAT))

        except:

            # Caso ocorra algum erro remove o cliente
            remove_client(client)
            break

    # Fecha a conexão
    client.close()

# Função principal responsável por aceitar novos clientes
def receive():

    while True:

        # Aguarda uma nova conexão
        client, addr = server.accept()

        # Exibe o endereço do cliente conectado
        print(f"Connected with {str(addr)}")

        # Solicita o nickname ao cliente
        client.send("NICK".encode(FORMAT))

        # Recebe o nickname
        nickname = client.recv(1024).decode(FORMAT)

        # Armazena o nickname
        nicknames.append(nickname)

        # Armazena o socket do cliente
        clients.append(client)

        # Exibe no servidor o nickname recebido
        print(f"Nickname of client is {nickname}")

        # Informa aos demais usuários que um novo cliente entrou
        broadcast(f"{nickname} joined the chat".encode(FORMAT))

        # Envia mensagem de confirmação ao cliente recém-conectado
        client.send("Connected to the server".encode(FORMAT))

        # Cria uma nova thread para atender esse cliente
        thread = threading.Thread(target=handle_client, args=(client, addr))

        # Inicia a thread
        thread.start()

# Mensagem indicando que o servidor foi iniciado
print("[STARTING] server is listening...")

# Inicia o servidor
receive()
